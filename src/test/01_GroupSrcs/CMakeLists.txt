GET_DIR_NAME(DIR_NAME)
SET(TARGET_NAME "${DIR_NAME}")

GLOBAL_GROUP_SOURCES(FILES0 ${CMAKE_CURRENT_SOURCE_DIR})
GLOBAL_GROUP_SOURCES(FILES1 "${PROJECT_SOURCE_DIR}/include/Test_01_GroupSrcs")
SET(SOURCES "")
LIST(APPEND SOURCES "${FILES0}")
LIST(APPEND SOURCES "${FILES1}")

SET(LIBS "")

ADD_TARGET(MODE "EXE" NAME "${TARGET_NAME}" SOURCES ${SOURCES} LIBS ${LIBS})
